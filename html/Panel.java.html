11   import java.awt.Container; 
12   import java.awt.GridLayout; 
13   import java.awt.event.ActionEvent; 
14   import java.awt.event.ActionListener;

19   /** 
20    * The UI class for a Whack a Mole simulation 
21    * Handles user interaction and updates game 
22    * according to inputs 
23    */ 
24   public class Panel extends JFrame implements ActionListener { 
25    
26       //Dimension of game board 
27       public static int dim; 
28    
29       //Represents the players current score 
30       public static int score; 
31    
32       //The amount of seconds the game will be played 
33       public static int time = 30; 
34    
35       //Labels for score and timer 
36       public static JLabel scoreLabel; 
37       public static JLabel timeLabel; 
38    
39       //The overall container the UI resides in 
40       public static Container pane; 
41    
42       //Array of buttons representing a mole thread 
43       public static MoleButton[][] moles; 
44    
45       public static Whacked game; 
46    
47       /** 
48        * Creates the game panel 
49        * Sets the layout and window configuration before generating 
50        * and configuring a button for each mole 
51        */ 
52       public Panel() { 
53           setTitle("Whack-a-Mole"); 
54           pane = getContentPane(); 
55    
56           //Acquire a board size from the player 
57           String input = JOptionPane.showInputDialog(pane, "Desired board dimension?"); 
58    
59           //Attempt to set the board size based on input, fail on invalid input and exit 
60           try { 
61               dim = Integer.parseInt(input); 
62           } catch (Exception e) { 
63               e.printStackTrace(); 
64               System.exit(1); 
65           } 
66    
67           pane.setLayout(new GridLayout(dim+1, dim, 5, 5)); 
68           moles = new MoleButton[dim][dim]; 
69    
70           //Generate a unique button for the number of moles 
71           for (int i = 0; i < dim; i++) { 
72               for (int j = 0; j < dim; j++) { 
73                   MoleButton button = new MoleButton(i, j); 
74                   pane.add(button); 
75                   button.addActionListener(this); 
76                   moles[i][j] = button; 
77               } 
78           } 
79    
80           //Add labels to bottom of window 
81           scoreLabel = new JLabel("Score: " + score); 
82           timeLabel = new JLabel("Time: " + time); 
83           pane.add(scoreLabel); 
84           pane.add(timeLabel); 
85       } 
86    
87       /** 
88        * The function which works during game play. Checks the 
89        * state of the game board 4 times per second and updates 
90        * the view accordingly. Also manages the user's score and 
91        * ends the game when the time has expired 
92        * @param game a reference to the game simulation 
93        * @throws InterruptedException 
94        */ 
95       public static void whack(Whacked game) throws InterruptedException { 
96           score = 0; 
97           int slept = 0; 
98           boolean playing = true; 
99           //Checks status of game board, sets the button labels based on state 
100          while (playing) { 
101              for (int i = 0; i < dim; i++) { 
102                  for (int j = 0; j < dim; j++) { 
103                      int[][] state = game.getBoard(); 
104                      if (state[i][j] == 1) { 
105                          moles[i][j].up(); 
106                      } else { 
107                          moles[i][j].down(); 
108                      } 
109                  } 
110              } 
111              scoreLabel.setText("Score: " + score); 
112   
113              //Sleep for quarter second before repeating update 
114              TimeUnit.MILLISECONDS.sleep(250); 
115   
116              //Every 4 sleeps, 1 second, decrement the timer 
117              slept++; 
118              if (slept % 4 == 0) { 
119                  time--; 
120                  timeLabel.setText("Time: " + time); 
121              } 
122              if (time == 0) { 
123                  playing = false; 
124              } 
125          } 
126   
127          //Game is over, present the score and then exit 
128          JOptionPane.showMessageDialog(pane, "Game Over!\nYour score was: " + score); 
129          System.exit(0); 
130      }

132      /** 
133       * Private extension of JButton class to allow for additional 
134       * fields for active state and coordinates on the grid 
135       */ 
136      private static class MoleButton extends JButton { 
137          public boolean up; 
138          public int x, y; 
139   
140          /** 
141           * Construct a mole button 
142           * @param x row coordinate on grid 
143           * @param y column coordinate on grid 
144           */ 
145          public MoleButton(int x, int y) { 
146              super(); 
147              this.up = false; 
148              this.setText(""); 
149              this.x = x; 
150              this.y = y; 
151          } 
152   
153          /** 
154           * Set state of button to up; mole is active 
155           */ 
156          public void up() { 
157              this.up = true; 
158              this.setText("UP!"); 
159          } 
160   
161          /** 
162           * Set state of button to down; mole is hiding 
163           */ 
164          public void down() { 
165              this.up = false; 
166              this.setText(""); 
167          } 
168   
169          /** 
170           * Check status of this mole 
171           * @return whether or not the mole is up or down 
172           */ 
173          public boolean check() { 
174              return this.up; 
175          } 
176      } 
177   
178      /** 
179       * Creates the game panel as well as a game simulation object 
180       * Begins execution of game after UI is drawn 
181       * @param args user program arguments 
182       */ 
183      public static void main(String[] args) { 
184          Panel gt = new Panel(); 
185          gt.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); 
186          gt.pack(); 
187          gt.setVisible(true); 
188          game = new Whacked(dim); 
189          game.startGame(); 
190          try { 
191              whack(game); 
192          } catch (InterruptedException e) { 
193              e.printStackTrace(); 
194          } 
195      } 
196   
197      /** 
198       * Handles the user input of a mouse click on a button 
199       * Checks if the button is in an up state and updates text 
200       * and score accordingly 
201       * @param e the captured mouse click event 
202       */ 
203      public void actionPerformed(ActionEvent e) { 
204          MoleButton mb = (MoleButton)e.getSource(); 
205          if (game.checkMoleThread(mb.x, mb.y)) { 
206              if (mb.check()) { 
207                  mb.down(); 
208                  score++; 
209              } 
210          } 
211      }
